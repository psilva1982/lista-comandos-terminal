# Comandos Git

# Iniciando um novo repositório
git init 

# Para não incluir um diretório no versionamento é necessário criar um 
# arquivo .gitignore na raiz do projeto e incluir o caminho do diretório dentro 
# do arquivo

## REMOVER ARQUIVO REMOTO
git pull origin master

git rm -r Arquivo Pasta

git commit -m "Remove Arquivo e Pasta"

git push origin master 

# Sincronizar o repositório local a partir do repositório remoto 
git pull
